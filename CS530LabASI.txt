placeholders 

date 
month
year
leapYear true or false
maxDate

START
	year = 0
	leapYear = false
	month = 0
	maxDate = 0
	date = 0

	DISPLAY "Please enter the year: "

	DO
		Get year

		IF year < 1582
			DISPLAY "Please enter a year that is valid to the Gregorian calendar"

		ELSE IF year is evenly divisible by 4 or year is evenly divisible by 100 or year is evenly divisible by 400 THEN
			leapYear = true

		END IF

	WHILE year < 1582 // the start of Gregorian calendar

	DISPLAY "Please enter the month: "

	DO
		GET month

		IF month > 13 or month < 0
			DESPLAY "Please enter a valid value"

		ELSE IF month = 4 or 6 or 9 or 11 THEN
			maxDate = 30

		ELSE IF month = 2 THEN
			
			IF leapYear = true
				maxDate = 29
			ELSE 
				maxDate = 28
			END IF 

		ELSE 
			maxDate = 31  

	WHILE month < 0 and month > 13

	DISPLAY "Please enter the date: "

	DO
		GET date

		IF date < 0 and date > maxDate THEN
			DISPLAY "Please enter a valid date"

		ELSE IF date = 1 and month > 1 THEN
				month <--- month - 1

        ELSE IF date = 1 and month = 1 THEN
        		month = 12

		END IF

	WHILE date < 0 and date > maxDate


	IF month = 4 or 6 or 9 or 11  THEN
		maxDate = 30

	ELSE IF month = 2 THEN
			
			IF leapYear = true
				maxDate = 29
			ELSE 
				maxDate = 28
			END IF

	ELSE 
		maxDate =31

	END IF



	IF date > 1 THEN

		previousYear <--- year
		previousMonth <--- month
		previousDate <--- date - 1

	ELSE IF date = 1 and month = 1  THEN

			previousYear <--- year - 1
			previousMonth <--- month
			previousDate <--- maxDate
	
	ELSE IF date = 1 and month < 1 THEN

			previousYear <--- year 
			previousMonth <--- month
			previousDate <--- maxDate

	END IF
END
